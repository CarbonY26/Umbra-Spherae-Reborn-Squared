# Visigothic province converts to Aragonese
province_event = {
	id = NC.1
	desc = EVTDESC_NC_1
	picture = GFX_evt_stone_church

	trigger = {
		should_evolve_cultures_trigger = yes
		culture = visigothic
		NOT = { religion_group = muslim }
		OR = {
			region = custom_aragonese
			owner = { culture = aragonese }
		}
		any_province_holding = {
			NOT = { holding_type = nomad }
		}
		NOT = {
			owner = { 
				top_liege = { 
					culture = visigothic 
					OR = {
						tier = emperor
						num_of_king_titles = 2
					}
				} 
			}
		}
		NOT = {
			owner = { 
				religion_head = { culture = visigothic }
			}
		}
		year = 850
	}
	
	mean_time_to_happen = {
		months = 3600
		modifier = {
			factor = 0.5
			owner = {
				culture = aragonese
			}
		}
		modifier = {
			factor = 0.3
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = EVTOPTA_NC_1
		culture = aragonese
	}
}

	id = NC.8
	desc = EVTDESC_NC_8
	picture = GFX_evt_viking_throneroom_oldgods

	trigger = {
		should_evolve_cultures_trigger = yes
		year = 950
		culture = norse
		region = custom_icelandic

		NOT = {
			owner = { 
				top_liege = { 
					culture = norse 
					OR = {
						tier = emperor
						num_of_king_titles = 2
					}
				} 
			}
		}
		NOT = {
			owner = { 
				religion_head = { culture = norse }
			}
		}
		OR = {
			AND = {
				owner = { top_liege = { culture = norse } }
				region = custom_icelandic
			}
			owner = { top_liege = { culture = icelandic } }
		}
	}
	
	mean_time_to_happen = {
		months = 2800
		modifier = {
			factor = 0.3
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = EVTOPTA_NC_8
		trigger = {
			NOT = {
				owner = { culture = icelandic }
			}
		}
		culture = icelandic
	}
	option = {
		name = EVTOPTB_NC_8

		should_evolve_cultures_trigger = yes

					culture = polish
					OR = {
						tier = emperor
						num_of_king_titles = 2
					}
				} 
			}
		}
		NOT = {
			owner = { 
				religion_head = { culture = polish }
			}
		}
		owner = {
			OR = {
				culture = silesian				
                                NOT = { 
                                      owner = {
                                              top_liege = { 
					      culture = polish
                                              } 
					}
                                }

					religion_group = pagan_group

						religion_head = { culture = THIS }

			}
		}
		any_province_holding = {
			NOT = { holding_type = nomad }
		}
	}
	
	mean_time_to_happen = {
		months = 1200
		modifier = {
			factor = 0.3
